from flask import Flask, render_template, request from datetime import datetime app = Flask(__name__) @app.route('/', methods=['GET', 'POST']) def index(): if request.method == 'POST': return render_template( 'result.html', teacher=request.form['teacher'], subject=request.form['subject'], school=request.form['school'], principal=request.form['principal'], program=request.form['program'], description=request.form['description'], date=datetime.now().strftime('%Y/%m/%d') ) return render_template('index.html')
